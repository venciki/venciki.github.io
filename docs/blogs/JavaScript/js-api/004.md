## 1. 简易版

```js
JSON.pares(JSON.stringfy())
```

此 api 能覆盖大多数场景，但实际上在某些严格场景，此方法有坑：

1. 无法解决循环引用问题，举个例子：

```js
const a = {val: 2};
a.target = a;
```

拷贝 a 会出现系统栈溢出，因为出现了无限递归的情况。

2. 无法拷贝一些特殊对象，比如 RegExp，Date，Set，Map等。

3. 无法拷贝函数

先写一个简易版的：
```js
function deepClone(target) {
    if (typeof target === 'object') {
        const resultObj = Array.isArray(target) ? [] : {};
        for (let props in target) {
            if (target.hasOwnProperty(props)) {
                resultObj[props] = deepClone(target[props]);
            }
        }
        return resultObj;
    }
    else {
        return target;
    }
}
```

## 2. 解决循环引用
问题：
```js
let obj = {a: 123};
obj.target = obj;

deepClone(obj); // 报错 RangeError: Maximum call stack size exceeded
```

如何解决：

创建一个 Map。记录下已经拷贝过的对象，如果已经拷贝过，直接返回即可。


```js
const isObject = (target) => (typeof target === 'object' || typeof target === 'function') && target !== null;

const deepClone = (target, map = new Map()) => {
    if (map.get(target)) {
        return target;
    }
    if (isObject(target)) {
        map.set(target, true);
        const resultObj = Array.isArray(target) ? [] : {};
        for (let props in target) {
            resultObj[props] = deepClone(resultObj[props]);
        }
        return resultObj;
    }
    else {
        return target;
    }
}
```
测试：
```js
let obj = {a: 123};
obj.target = obj;

console.log(deepClone(obj));
// { a: 123, target: <ref *1> { a: 123, target: [Circular *1] } }
```

看着是没啥问题了，但还有一个潜在的坑，就是 map 上的 key 和 map 构成了`强引用关系`。

被弱引用的对象可以在任何时候被回收，对于强引用来说，只要强引用还在，那么对象就无法被回收。拿上面的例子来说，map 和 obj 一直是强引用关系，在程序结束之前，obj 所占的内存空间一直不会被释放。

如何解决？

让 map 的 key 和 map 构成弱引用即可。ES6给我们提供了这样的数据结构，它的名字叫 `WeakMap` ，它是一种特殊的 Map, 其中的键是弱引用的。其键必须是对象，而值可以是任意的。

稍加改造：

```js
const deepClone = (target, map = new WeakMap()) => {
  //...
}
```
## 3. 拷贝特殊对象

### 可遍历对象

对于特殊对象，我们使用以下方式来鉴别：
```js
Object.prototype.toString.call(obj);
```

梳理一下可遍历对象：

```js
["object Map"]
["object Set"]
["object Array"]
["object Object"]
["object Arguments"]
```

根据这些不同的字符串，就可以鉴别

```js
const isObject = (target) => (typeof target === 'object' || typeof target === 'function') && target !== null;

const getType = (obj) => Object.prototype.toString.call(obj);

const canTraverse = {
  '[object Map]': true,
  '[object Set]': true,
  '[object Array]': true,
  '[object Object]': true,
  '[object Arguments]': true,
};

const mapTag = '[object Map]';
const setTag = '[object Set]';

const deepClone = (target, map = new Map()) => {
    if (!isObject(target)) {
        return target;
    }

    let type = getType(target);
    let resultObj;
    if (!canTraverse[type]) {
        // 处理不可遍历对象
        return;
    }
    else {
        // 这波操作相当关键，可以保证对象的原型不丢失！
        let ctor = target.prototype;
        resultObj = new ctor();
    }

    if (map.get(target)) {
        return target;
    }

    map.set(target, true);

    // 处理Map
    if (type === mapTag) {
        target.forEach((item , key)=> {
            resultObj.set(deepClone(key), deepClone(item));
        })
    }

    // 处理Set
    if (type === setTag) {
        target.forEach(item=> {
            resultObj.add(deepClone(item));
        })
    }

    // 处理数组和对象
    for (let prop in target) {
        if (target.hasOwnProperty(prop)) {
            resultObj[prop] = deepClone(target[prop]);
        }
    }

    return resultObj;

}
```

### 不可遍历对象

```js
const boolTag = '[object Boolean]';
const numberTag = '[object Number]';
const stringTag = '[object String]';
const dateTag = '[object Date]';
const errorTag = '[object Error]';
const regexpTag = '[object RegExp]';
const funcTag = '[object Function]';
```
对于不可遍历的对象，不同对象有不同的处理方法。

```js
const handleRegExp = (target) => {
    const {source, flags} = target;
    return new target.constructor(source, flags);
}

const handleFunc = (target) => {
    // ...
}

const handleNotTraverse = (target, tag) => {
    const Ctor = target.constuctor;
    switch (tag) {
        case boolTag:
        case numberTag:
        case stringTag:
        case errorTag:
        case dateTag:
            return new Ctor(target);
        case regexpTag:
            return handleRegExp(target);
        case funcTag:
            return handleFunc(target);
        default:
            return new Ctor(target);
    }
}
```
## 4. 拷贝函数
提到函数，在JS种有两种函数，一种是普通函数，另一种是箭头函数。每个普通函数都是 Function的实例，而箭头函数不是任何类的实例，每次调用都是不一样的引用。那我们只需要 处理普通函数的情况，箭头函数直接返回它本身就好了。

那么如何来区分两者呢？

答案是: 利用原型。箭头函数是不存在原型的。

代码如下:

```js
const handleFunc = (func) => {
    // 箭头函数直接返回自身
    if(!func.prototype) return func;
    const bodyReg = /(?<={)(.|\n)+(?=})/m;
    const paramReg = /(?<=\().+(?=\)\s+{)/;
    const funcString = func.toString();
    // 分别匹配 函数参数 和 函数体
    const param = paramReg.exec(funcString);
    const body = bodyReg.exec(funcString);
    if(!body) return null;

    if (param) {
        const paramArr = param[0].split(',');
        return new Function(...paramArr, body[0]);
    } else {
        return new Function(body[0]);
    }
}
```

至此深拷贝已经很完善了


## 5. bug

```js

```

对于这样一个bug，我们可以对 Boolean 拷贝做最简单的修改， 调用valueOf: new target.constructor(target.valueOf())。

6. 完整代码

```js
const isObject = (target) => (typeof target === 'object' || typeof target === 'function') && target !== null;

const getType = (obj) => Object.prototype.toString.call(obj);

const canTraverse = {
  '[object Map]': true,
  '[object Set]': true,
  '[object Array]': true,
  '[object Object]': true,
  '[object Arguments]': true,
};

const mapTag = '[object Map]';
const setTag = '[object Set]';
const boolTag = '[object Boolean]';
const numberTag = '[object Number]';
const stringTag = '[object String]';
const dateTag = '[object Date]';
const errorTag = '[object Error]';
const regexpTag = '[object RegExp]';
const funcTag = '[object Function]';

const handleRegExp = (target) => {
    const {source, flags} = target;
    return new target.constructor(source, flags);
}

const handleFunc = (func) => {
    // 箭头函数直接返回自身
    if(!func.prototype) return func;
    const bodyReg = /(?<={)(.|\n)+(?=})/m;
    const paramReg = /(?<=\().+(?=\)\s+{)/;
    const funcString = func.toString();
    // 分别匹配 函数参数 和 函数体
    const param = paramReg.exec(funcString);
    const body = bodyReg.exec(funcString);
    if(!body) return null;

    if (param) {
        const paramArr = param[0].split(',');
        return new Function(...paramArr, body[0]);
    } else {
        return new Function(body[0]);
    }
}


const handleNotTraverse = (target, tag) => {
    const Ctor = target.constructor;
    switch(tag) {
        case boolTag:
            return new Object(Boolean.prototype.valueOf.call(target));
        case numberTag:
            return new Object(Number.prototype.valueOf.call(target));
        case stringTag:
            return new Object(String.prototype.valueOf.call(target));
        case errorTag: 
        case dateTag:
            return new Ctor(target);
        case regexpTag:
            return handleRegExp(target);
        case funcTag:
            return handleFunc(target);
        default:
            return new Ctor(target);
    }
}


const deepClone = (target, map = new Map()) => {
    if (!isObject(target)) {
        return target;
    }

    let type = getType(target);
    let resultObj;
    if (!canTraverse[type]) {
        // 处理不可遍历对象
        return handleNotTraverse(target, type);
    }

    if (map.has(target)) {
        return map.get(target);
    }

    // 这波操作相当关键，可以保证对象的原型不丢失！
    let Ctor = target.constructor;
    resultObj = new Ctor();

    map.set(target, resultObj);

    // 处理Map
    if (type === mapTag) {
        target.forEach((item , key)=> {
            resultObj.set(deepClone(key, map), deepClone(item, map));
        })
        return resultObj;
    }

    // 处理Set
    if (type === setTag) {
        target.forEach(item=> {
            resultObj.add(deepClone(item, map));
        })
        return resultObj;
    }

    // 处理数组和对象
    for (let prop in target) {
        if (target.hasOwnProperty(prop)) {
            resultObj[prop] = deepClone(target[prop], map);
        }
    }

    return resultObj;
}
```
