```js
Function.prototype.call = function(context, ...args) {
    const ctx = context || window;
    ctx['fn'] = this;
    const result = ctx.fn(...args);
    delete ctx['fn'];
    return result;
}
```


```js
Function.prototype.apply = function(context, args) {
    const ctx = context || window;
    ctx['fn'] = this;
    const result = ctx.fn(...args);
    delete ctx['fn'];
    return result
}
```
