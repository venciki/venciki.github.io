var jump = function(nums) {
    if (nums.length <= 1) {
        return 0;
    }

    let step = 0;
    let maxStep = 0;
    let lastMaxStep = 0;

    for (let i = 0; i < nums.length; i++) {
        maxStep = Math.max(maxStep, nums[i] + i);
        if (maxStep >= nums.length - 1) {
            step++;
            break;
        }
        else if (i === lastMaxStep) {
            step++;
            lastMaxStep = maxStep;
        }
    }

    return step;
}
