var rob = function(nums) {
    const len = nums.length;
    const dp = Array(len);

    dp[0] = nums[0];
    dp[1] = Math.max(nums[0], nums[1]);

    for (let i = 2; i < len; i++) {
        // dp[i - 1] + 0： 当前不偷的结果
        // dp[i - 2] + nums[i]：表示第i个偷的情况下的结果
        dp[i] = Math.max(dp[i - 1] + 0, dp[i - 2] + nums[i]);
    }

    return dp[len - 1];
};
