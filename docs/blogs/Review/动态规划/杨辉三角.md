var generate = function(numRows) {
    const c = Array(numRows);
    for (let y = 0; y < numRows; y++) {
        c[y] = Array(y + 1);
        c[y][0] = 1;
        c[y][y] = 1;

        for (x = 1; x < y; x++) {
            c[y][x] = c[y - 1][x -1] + c[y - 1][x];
        }
    }
    return c;
}
